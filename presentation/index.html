Remx Basics

Create Store

                
                  import * as remx from 'remx';
                  const initialState = {
                    loading: true,
                    posts: {},
                    selectedPosts: [],
                  };
                  const state = remx.state(initialState);

Create Setters & getters

                
                  import * as remx from 'remx';
    
                  export const getters = remx.getters({
                    isLoading() {
                      return state.loading;
                    }
                  });
    
                  export const setters = remx.setters({
                    setLoading(isLoading) {
                      state.loading = isLoading;
                    }   
                 });

Connect store to component

                
                    import { connect } from 'remx';
                    import {getters as store} from './someStore';
                    
                    const SomeComponent = (props) => (
                      {props.isLoading}
                    );
                    
                    const mapStateToProps = (ownProps) => ({
                      isLoading: store.isLoading();
                    });
                    
                    export default connect(mapStateToProps)(SomeComponent);

Recompose

HOC lib
composable
allows to split logic and view

Example #1

                
                    import {compose, withState, mapProps, pure} from 'recompose';
                    const BaseComponent = props => {...}
                    
                    // Note that the order has reversed â€” props flow from top to bottom
                    const enhance = compose(
                      withState(/*...args*/),
                      mapProps(/*...args*/),
                      pure
                    )

                    const EnhancedComponent = enhance(BaseComponent)

Example #2

                
                    import {compose, branch, renderComponent} from 'recompose';
                    const BaseComponent = props => {...}
                    const LoaderComponent = props => {...}

                    const spinnerWhileLoading = isLoadingPred => branch(
                      isLoadingPred,
                      renderComponent(LoaderComponent)
                    );

                    const enhance = spinnerWhileLoading(props => props.isLoading);
                    //...
                    export const FinalComponent = compose(
                      connect(someAppStateProps),
                      enhance
                    )(BaseComponent)
